## 2. Multiple API Keys Strategy (Key Rotation)

Yes, this works technically — a key rotation array in your code that cycles to the next key when one hits 429. But **a heads-up**: using multiple Google accounts to bypass rate limits technically violates Google's ToS. You probably won't get banned for a student project, but know the risk exists.

Here's how to implement it cleanly:

```javascript
// keyRotator.js
const GEMINI_KEYS = [
  process.env.GEMINI_KEY_1,
  process.env.GEMINI_KEY_2,
  process.env.GEMINI_KEY_3,
  // add up to 10
].filter(Boolean);

let currentIndex = 0;
const keyFailCounts = {};

export function getNextKey() {
  // Round-robin with skip on exhausted keys
  for (let i = 0; i < GEMINI_KEYS.length; i++) {
    const key = GEMINI_KEYS[currentIndex % GEMINI_KEYS.length];
    currentIndex++;
    if ((keyFailCounts[key] || 0) < 3) return key;
  }
  // All keys exhausted — reset and try again
  Object.keys(keyFailCounts).forEach(k => keyFailCounts[k] = 0);
  return GEMINI_KEYS[0];
}

export function markKeyFailed(key) {
  keyFailCounts[key] = (keyFailCounts[key] || 0) + 1;
}
```

```javascript
// In your Gemini caller
async function callGemini(prompt, retries = 0) {
  const key = getNextKey();
  try {
    const res = await fetch(`https://generativelanguage.googleapis.com/...?key=${key}`, ...);
    if (res.status === 429) {
      markKeyFailed(key);
      if (retries < GEMINI_KEYS.length) return callGemini(prompt, retries + 1);
      throw new Error("All keys rate-limited");
    }
    return res.json();
  } catch (err) {
    throw err;
  }
}
```
